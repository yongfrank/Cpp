163bool match(std::string &pattern, std::string &source) {
164 auto positionOfQuestionMark = pattern.find("?");
165 auto positionOfAsteriskMark = pattern.find("*");
166 if(positionOfQuestionMark != std::string::npos) {
167 pattern.replace(positionOfQuestionMark, 1, ".");
168 }
169 if(positionOfAsteriskMark != std::string::npos) {
170 pattern.replace(positionOfAsteriskMark, 1, ".*");
171 }
172
173 return std::regex_match(source, std::regex(pattern));
174 // return (source.find(pattern) != std::string::npos) ? true : false;
175}
