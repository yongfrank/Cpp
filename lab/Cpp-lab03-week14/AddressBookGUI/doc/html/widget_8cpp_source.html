322void Widget::initModelFromStringList(QStringList& fileContentStringList)
323{
324 // 将内存中的 model 数据模型清除
325 this->model->clear();
326
327 // 从一个StringList 获取数据，初始化数据Model
328 // 文本行数，第1行是标题
329 int rowCount = fileContentStringList.count();
330
331 // 实际数据行数
332 this->model->setRowCount(rowCount - 1);
333
334 // 设置表头
335 // 第1行是表头
336 QString header = fileContentStringList.at(0);
337
349 static auto multiTab = QRegularExpression("\\t+");
350
351 // 一个或多个空格、TAB("\s+") 多个 TAB 分隔("\t+")等分隔符隔开的字符串， 分解为一个StringList
352 QStringList headerList = header.split(multiTab);
353 headerList.removeLast();
354 //设置表头文字
355 this->model->setHorizontalHeaderLabels(headerList);
356
357 //设置表格数据
358// QString lineText;
359// QStringList lineTempList;
360 QStandardItem *itemStandard;
361 for(int i = 0; i < rowCount; i++) {
362
363 //获取数据区的一行
364 QString lineText = fileContentStringList.at(i);
365
366 //一个或多个空格、TAB等分隔符隔开的字符串， 分解为一个StringList
367 QStringList lineTempList = lineText.split(multiTab);
368
369 //lineTempList 的行数等于 headerList.count, 固定的
370 for(int j = 0; j < headerList.count(); j++) {
371
372 //创建item
373 itemStandard = new QStandardItem(lineTempList.at(j));
374
375 //为模型的某个行列位置设置Item
376 this->model->setItem(i - 1, j, itemStandard);
377 }
378 }
379 this->customizedTableStyle();
380}
