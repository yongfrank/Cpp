146bool match(std::string &pattern, std::string &source) {
147 auto positionOfQuestionMark = pattern.find("?");
148 auto positionOfAsteriskMark = pattern.find("*");
149 if(positionOfQuestionMark != std::string::npos) {
150 pattern.replace(positionOfQuestionMark, 1, ".");
151 }
152 if(positionOfAsteriskMark != std::string::npos) {
153 pattern.replace(positionOfAsteriskMark, 1, ".*");
154 }
155
156 return std::regex_match(source, std::regex(pattern));
157 // return (source.find(pattern) != std::string::npos) ? true : false;
158}
